namespace = situation_operation

@StealResourcesTimer = 10

## Steal Resources, Stage 1 ##
espionage_operation_event = {
	id = situation_operation.1100
	title = situation_operation.1100.name
	desc = situation_operation.1100.desc
	picture = GFX_evt_sabotaged_ship
	show_sound = event_default
	is_triggered_only = yes
	espionage_operation = yes

	option = {
		name = UNFORTUNATE
		trigger = {
			NOT = { exists = event_target:easy }
			NOT = { exists = event_target:norm }
			NOT = { exists = event_target:hard }
		}
		destroy_espionage_operation = this
	}

	option = {
		name = situation_operation.1100.option.easy
		trigger = { exists = event_target:easy }
		hidden_effect = {
			event_target:easy = {
				save_event_target_as = operation_target
			}
		}
	}

	option = {
		name = situation_operation.1100.option.norm
		trigger = { exists = event_target:norm }
		hidden_effect = {
			event_target:norm = {
				save_event_target_as = operation_target
			}
		}
	}

	option = {
		name = situation_operation.1100.option.hard
		trigger = { exists = event_target:hard }
		hidden_effect = {
			event_target:hard = {
				save_event_target_as = operation_target
			}
		}
	}

	immediate = {
		# Stop operations rolls until option is picked.
		set_espionage_operation_progress_locked = yes
		# Select planets to give as options to the player.
		from = {
			random_planet_within_border = {
				limit = {
					is_owned_by = from
					num_pops <= 25
				}
				save_event_target_as = easy
			}
			random_planet_within_border = {
				limit = {
					is_owned_by = from
					OR = {
						NOT = { exists = event_target:easy }
						NOT = { is_planet = event_target:easy }
					}
					num_pops > 25
					num_pops < 75
				}
				save_event_target_as = norm
			}
			random_planet_within_border = {
				limit = {
					is_owned_by = from
					OR = {
						NOT = { exists = event_target:easy }
						NOT = { is_planet = event_target:easy }
					}
					OR = {
						NOT = { exists = event_target:norm }
						NOT = { is_planet = event_target:norm }
					}
					num_pops >= 75
				}
				save_event_target_as = hard
			}
		}
	}

	after = {
		event_target:operation_target = {
			# Set flags on the target planet.
			hidden_effect = {
				set_planet_flag = operation_resource_target@root
				set_planet_flag = operation_resource_target_global
			}
			# Start the situation on the target planet.
			owner = {
				start_situation = {
					type = situation_espionage_resources
					target = event_target:operation_target
					effect = {
						# Set flags for going between the situation and operation.
						set_situation_flag = operation_resource_situation@root
						root = { set_espionage_operation_flag = operation_resource_attacker@prev }
					}
				}
			}
		}
		# Start operation rolls.
		set_espionage_operation_progress_locked = no
	}
}

## Steal Resources, Result Stage ##
espionage_operation_event = {
	id = situation_operation.1900
	title = situation_operation.1900.name
	desc = situation_operation.1900.desc
	picture = GFX_evt_sabotaged_ship
	show_sound = event_espionage_concluded
	is_triggered_only = yes

	option = {
		name = situation_operation.1900.option.take
	}

	option = {
		name = situation_operation.1900.option.destroy
	}

	immediate = {
		# Save flagged planet as event scope for localization.
		from = {
			random_owned_planet = {
				limit = { has_planet_flag = operation_resource_target@root }
				save_event_target_as = selected_planet
			}
			random_situation = {
				limit = { has_situation_flag = operation_resource_situation@root }
				save_event_target_as = espionage_sitation
			}
		}
		if = {
			limit = {
				OR = {
					NOT = { exists = event_target:espionage_sitation }
					event_target:espionage_sitation = { situation_progress >= 80 }
				}
			}
			# Situation ended at Stage 5 or Completed.
			set_variable = { which = situation_multiplier value = 0.0 }
		}
		else_if = {
			limit = {
				event_target:espionage_sitation = { situation_progress >= 60 }
			}
			# Situation ended at Stage 4.
			set_variable = { which = situation_multiplier value = 0.2 }
		}
		else_if = {
			limit = {
				event_target:espionage_sitation = { situation_progress >= 40 }
			}
			# Situation ended at Stage 3.
			set_variable = { which = situation_multiplier value = 0.5 }
		}
		else_if = {
			limit = {
				event_target:espionage_sitation = { situation_progress >= 20 }
			}
			# Situation ended at Stage 2.
			set_variable = { which = situation_multiplier value = 0.8 }
		}
		else = {
			# Situation ended at Stage 1.
			set_variable = { which = situation_multiplier value = 1.0 }
		}
	}

	after = {
		# Delay repeating this operation.
		spynetwork = {
			set_saved_date = {
				key = recently_stole_resources
				days_from_present = @StealResourcesTimer
				expires = @StealResourcesTimer
			}
		}
		# Add resources based on situation multiplier.
		root.owner = {
			add_resource = {
				minerals = 100
				mult = event_target:espionage_sitation.trigger:situation_progress
			}
		}
		# End this operation.
		destroy_espionage_operation = this
	}
}
